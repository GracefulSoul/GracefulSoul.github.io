    class Solution {

  public int[] fairCandySwap(int[] aliceSizes, int[] bobSizes) {
    int diff = 0;
    Set<Integer> set = new HashSet<>();
    for (int aliceSize : aliceSizes) {
      diff += aliceSize;
      set.add(aliceSize);
    }
    for (int bobSize : bobSizes) {
      diff -= bobSize;
    }
    diff /= 2;
    for (int bobSize : bobSizes) {
      if (set.contains(bobSize + diff)) {
        return new int[] { bobSize + diff, bobSize };
      }
    }
    return new int[0];
  }

}
