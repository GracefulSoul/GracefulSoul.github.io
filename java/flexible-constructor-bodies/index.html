    import java.util.regex.Pattern;

public class ConfigurationClass {
    
    private Pattern validationPattern;
    private String ipAddress;
    private int port;
    
    public ConfigurationClass(String pattern, String ipAddress, int port) {
        // н”„лЎӨлЎңк·ё: ліөмһЎн•ң кі„мӮ° л°Ҹ кІҖмҰқ
        if (port < 1 || port > 65535) {
            throw new IllegalArgumentException(
                "нҸ¬нҠёлҠ” 1-65535 мӮ¬мқҙм—¬м•ј н•©лӢҲлӢӨ"
            );
        }
        
        // м •к·ңн‘ңнҳ„мӢқ м»ҙнҢҢмқј - мӢӨнҢЁн•  мҲҳ мһҲмқҢ
        Pattern compiled = compilePattern(pattern);
        
        // IP мЈјмҶҢ кІҖмҰқ
        validateIpAddress(ipAddress);
        
        // ліөмһЎн•ң кі„мӮ°мқ„ нҶөн•ң мҙҲкё°к°’ м„Өм •
        this.validationPattern = compiled;
        this.ipAddress = ipAddress;
        this.port = port;
        
        System.out.println("м„Өм •мқҙ м„ұкіөм ҒмңјлЎң мҙҲкё°нҷ”лҗҳм—ҲмҠөлӢҲлӢӨ");
    }
    
    private static Pattern compilePattern(String pattern) {
        try {
            return Pattern.compile(pattern);
        } catch (Exception e) {
            throw new IllegalArgumentException(
                "мң нҡЁн•ҳм§Җ м•ҠмқҖ м •к·ңн‘ңнҳ„мӢқ: " + pattern
            );
        }
    }
    
    private static void validateIpAddress(String ip) {
        String[] parts = ip.split("\\.");
        if (parts.length != 4) {
            throw new IllegalArgumentException(
                "мң нҡЁн•ҳм§Җ м•ҠмқҖ IP мЈјмҶҢ: " + ip
            );
        }
        
        for (String part : parts) {
            try {
                int num = Integer.parseInt(part);
                if (num < 0 || num > 255) {
                    throw new NumberFormatException();
                }
            } catch (NumberFormatException e) {
                throw new IllegalArgumentException(
                    "к°Ғ мҳҘн…ҹмқҖ 0-255 мӮ¬мқҙм—¬м•ј н•©лӢҲлӢӨ"
                );
            }
        }
    }
    
    public static void main(String[] args) {
        try {
            ConfigurationClass config = new ConfigurationClass(
                "^[a-zA-Z0-9]+$",
                "192.168.1.1",
                8080
            );
            
            ConfigurationClass invalid = new ConfigurationClass(
                "[invalid",
                "192.168.1.1",
                8080
            );
            
        } catch (IllegalArgumentException e) {
            System.err.println("мҳӨлҘҳ: " + e.getMessage());
        }
    }
}
